Distances Between Subspaces

This question came up when trying to compare kernels as well as how low dimensional projections change over time. Intuitively it isn’t clear how to create a distance metric between subspaces. Code 

import torch
import math

def grassman(A, B):
		# get orthogonal basis
    Q_A, R = torch.linalg.qr(A)
    Q_B, R = torch.linalg.qr(B)

		# get singular values
    _, S, _  = torch.svd(Q_A.T @ Q_B)
    # cosine inverse
    thetas = torch.arccos(S)
    # return norm
    return torch.norm(thetas, p=2)
A = torch.tensor([
    [1, 0],
    [0, 1],
    [0, 0]
]).float()
B = torch.tensor([
    [math.sqrt(0.5), 0],
    [0, 1],
    [math.sqrt(0.5), 0]
]).float()
C = torch.tensor([
    [0, 0],
    [1, 0],
    [0, 1]
]).float()
>>> grassman(A,B)
tensor(0.7854)
>>> grassman(A,C)
tensor(1.5708)
>>> grassman(C,B)
tensor(0.7854)
